\section*{Постановка задачи. Основные уравнения}

Для описания динамики ледника используем несколько уравнений
\begin{itemize}
    \item закон сохранения массы
        \begin{equation*}
        \nabla
        \cdot
        \vec{v}
        =
        0
        \end{equation*}
    \item закон сохранения момента
        \begin{equation*}
            \rho
            \frac
                {d \vec{v}}
                {dt}
            =
            \nabla
            \cdot
            \sigma
            +
            \rho
            \vec{g}
        \end{equation*}
    \item закон сохранения энергии
        \begin{equation*}
            \rho
            \frac
                {d \left( c_{p}\Theta \right)}
                {dt}
            =
            \nabla
            \left(
                k_{i}
                \nabla
                \Theta
            \right)
            +
            \Phi
        \end{equation*}
\end{itemize}

Дополнительные уравнения
\begin{itemize}
    \item уравнение поверхности
    \begin{equation*}
        \nabla
        \cdot
        H
        \overline{v_{H}}
        =
        \dot{M}_s
        -
        \dot{M}_b
        -
        \frac
            {\partial H}
            {\partial t}
    \end{equation*}
\end{itemize}
